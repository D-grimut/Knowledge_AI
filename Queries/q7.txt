SELECT ?lectureContent
WHERE{
?course uni:ID ?courseNumber.
?course uni:has_lecture ?lecture.
?lecture uni:has_content ?lectureContent.
?lectureContent rdf:type uni:OtherLectureMaterial.
FILTER(?courseNumber=474).
}
LIMIT 100
