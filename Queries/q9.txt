WHERE{
?coures uni:subject ?cName.
?course uni:has_topic ?tName.
?course uni:has_lecture ?lecture.

FILTER(REGEX(STR(?cName), 'Introduction_to_Software_Engineering', "i")). 
}
LIMIT 100
